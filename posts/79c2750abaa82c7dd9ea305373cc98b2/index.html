从键盘输入的年份，判断该年是否是闰年

#include<stdio.h>
int main()
{
int y,l;
printf(“年份:”);
scanf(“%d”,&y);
if(y%4== 0)
{
if(y%100== 0)
{
if(y%400==0)
l=1;
else
l=0;
}
else
l=1;
}
else
l=0 ;
if(l)
printf(“%d年是闰年”,y);
else
printf(“%d年不是闰年”,y);
}

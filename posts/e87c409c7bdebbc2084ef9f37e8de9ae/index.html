#include<cstdio>
#include<iostream>
using namespace std;
const int P=998244353,M=510000;
int ans=0,g[M],inv[M],n,m,w[2][M],jc[M];
int abs(int x){return x>0?x:-x;}
int pow(int a,int k){int ret=1;for(;k;a=1ll*a*a%P,k>>=1) if(k&1)ret=1ll*ret*a%P;return ret;}
void pre(){
	int g=pow(3,(P-1)/n);w[0][0]=w[1][0]=1;
	for(int i=1;i<n;i++) w[0][i]=1ll*w[0][i-1]*g%P;
	for(int i=1;i<n;i++) w[1][i]=w[0][n-i];
}
int ntt(int *a,int f){
	for(int i=0,j=0;i<n;i++){
		if(i>j) swap(a[i],a[j]);
		for(int l=n>>1;(j^=l)<l;l>>=1);
	}
	for(int i=1;i<n;i<<=1)for(int j=0;j<n;j+=(i<<1))
	for(int k=0;k<i;k++){
		int x=a[k+j],y=1ll*w[f][n/(i<<1)*k]*a[i+k+j]%P;
		a[k+j]=(x+y)%P;a[i+j+k]=(x-y+P)%P;
	}	
	if(!f)for(int i=0,inv1=pow(n,P-2);i<n;i++) a[i]=1ll*a[i]*inv1%P;
}
int main(){
	scanf("%d",&n);
	inv[n]=1;inv[0]=1;jc[0]=1;
	for(int i=1;i<=n;i++)inv[n]=1ll*inv[n]*i%P,jc[i]=1ll*jc[i-1]*i%P;
	inv[n]=pow(inv[n],P-2);
	for(int i=n-1;i>=1;i--) inv[i]=1ll*inv[i+1]*(i+1)%P;
	for(int i=1;i<=n;i++) inv[i]=inv[i]*((i&1)?-1:1);
	g[0]=1;g[1]=n+1;
	for(int i=2;i<=n;i++){
		int t=(pow(i,n+1)-1+P)%P;
		t=1ll*t*pow(i-1,P-2)%P;
		g[i]=1ll*t*abs(inv[i])%P;
	}
	
	m=2*n;for(n=1;n<m;n<<=1);pre();
	ntt(inv,1);ntt(g,1);
	for(int i=0;i<n;i++) g[i]=1ll*g[i]*inv[i]%P;
	ntt(g,0);
	for(int i=0;i<=m/2;i++) ans=(ans+1ll*pow(2,i)*jc[i]%P*g[i]%P)%P;
	printf("%d",(ans+P)%P);
}

01背包和完全背包

01背包

int main()
{
	//01背包，time==》V    分数等于价值，时间等于体积
	int size; cin >> size;
	vector<int>  w; w.resize(size + 1, 0);
	vector<int>  v; v.resize(size + 1, 0);
	vector<int> dp;

	for (int i = 1; i <= size; i++)
	{
		int temp; cin >> temp;
		v[i] = temp;
	}
	for (int i = 1; i <= size; i++)
	{
		int temp; cin >> temp;
		w[i] = temp;
	}
	int weight = 10; //cin >> weight;
	dp.resize(weight + 1, 0);
	for (int i = 1; i <= size; i++)
	{
		for (int j = weight; j >= w[i]; j--)
		{
			dp[j] = max(dp[j], dp[j - w[i]] + v[i]);

		}
	}
	cout << dp[weight];
}

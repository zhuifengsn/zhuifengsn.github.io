esp32 oled初始化

# 导入相关模块
from pyb import ExtInt
from machine import I2C,Pin
from ssd1306 import SSD1306_I2C
# 初始化 OLED 模块
i2c = I2C(sda=Pin("Y8"), scl=Pin("Y6"))
oled = SSD1306_I2C(128, 64, i2c, addr=0x3c)
# OLED 初始信息显示
oled.fill(0)                                                                 # 清屏背景黑色
oled.text("Human body test:", 0, 15)          # 写入第1行内容
oled.show()                                                           # OLED 执行显示

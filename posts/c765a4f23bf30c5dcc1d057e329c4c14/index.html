01:判断数正负

#include<iostream>
using namespace std;
int main(){
	int n;
	scanf("%d",&n);
    if(n>0)
		cout<<"positive"<<endl;
	else if(n==0)
		cout<<"zero"<<endl;
	else 
		cout<<"negative"<<endl;
}

02:输出绝对值

#include<iostream>
#include<cmath>
using namespace std;
int main(){
	double n;
	cin>>n;
	if(n>=0)
		printf("%.2lf\n",n);
	else 
		printf("%.2lf\n",abs(n));
	return 0;
}

03:奇偶数判断

#include<iostream>
using namespace std;
int main(){
	int n;
	cin>>n;
	if(n%2==1)
		cout<<"odd";
	else 
		cout<<"even";
	return 0;
}

04:奇偶ASCII值判断

#include <iostream>  
using namespace std;  
int main(){  
    char ch;  
    ch = getchar();  
    if (ch%2 == 0) cout << "NO" << endl;  
    else cout << "YES" << endl;  
    return 0;  
}

05:整数大小比较

#include<iostream>
using namespace std;
int main(){
	int a,b;
	cin>>a>>b;
	if(a>b)cout<<'>';
	else if(a==b)cout<<'=';
	else cout<<'<';
	return 0;
}

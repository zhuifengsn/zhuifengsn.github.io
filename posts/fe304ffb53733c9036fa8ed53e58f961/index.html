STM32Cubmax key使用软件延时检查输入状态

一、key的硬件图

二、软件消抖原理

当按键按下会有些波动干扰，比如PE4被按下的时候，KEY0的电压就会被拉低，变为1，由于抖动，所以在有些时候电平会变为1，因此我们通过软件延时把开始时候的抖动过程给消除。

#include "main.h"
#define Key HAL_GPIO_ReadPin(GPIOE,GPIO_PIN_4)
//define Key HAL_GPIO_ReadPin(KEY_GPIO_Port,KEY_Pin)
#define LED HAL_GPIO_TogglePin(GPIOB,GPIO_PIN_5)
//define LED HAL_GPIO_TogglePin(LED_GPIO_Port,LED_Pin)


void Scan_Key(int mode)
{
	//模式1为按下就改变亮灭
	if(mode == 1)
	{
		if(Key == GPIO_PIN_RESET){  //检测跳变为低电平
			HAL_Delay(30); //消抖
			if(Key == GPIO_PIN_RESET){ //再次检测跳变为低电平，确定按键按下
				LED; //翻转电平
				while(!Key); //等待松手。即跳变为高电平，跳出循环
				//ps,如果不加这个循环，按下为低电平，这样就会一直改变LED电平
			}
		}
	}
}

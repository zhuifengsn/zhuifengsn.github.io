（Python十一讲）、Tuple数据类型元组

#Tuple数据类型元组
# GQ, 2022/01/04


tup1 = ('Google','Python3',100,200)
tup2 = (1,2,3,4,5)
tup3 = ('a','b','c')
tup4 = ()       #空元组
print(type(tup1))
print(tup1 + tup2)
print(tup4)
print(tup1[1])
print(tup1[1:])

#元组中只包含一个元素时，需要在元素后加,号分隔，否则会被当成运算符使用
tup5 = (50)
print(tup5)         #不加,号，类型为整型
tup6 = (50,)
print(tup6)         #加,号，类型为元组

#删除元组,  元组中的元素值是不允许删除的
tup7 = ('Google','Python3','GaoQiang',1000,2000)
print(tup7)
# del tup7[0]       #元组中的元素值是不允许删除的
print(tup7)

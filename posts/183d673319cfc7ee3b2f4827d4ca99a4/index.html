FSelect文件

下载 FSelect 将  fSelect.css 与 fSelect.js 拖入项目
需要引入 fSelect.css 与 fSelect.js 文件
<link href="fSelect.css" rel="stylesheet" type="text/css">
<script src="fSelect.js"></script>

<!doctype html>
<html>
<head>
    <meta charset="utf-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge,chrome=1">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <title>jQuery fSelect Plugin Demo</title>
    <link href="fSelect.css" rel="stylesheet" type="text/css">
</head>
<body>
    <select class="sel" multiple="multiple" style="width: 500px;">
    <optgroup label="Languages">
        <option value="cp">C++</option>
        <option value="cs">C#</option>
        <option value="oc">Object C</option>
        <option value="c">C</option>
    </optgroup>
    <optgroup label="Scripts">
        <option value="js">JavaScript</option>
        <option value="php">PHP</option>
        <option value="asp">ASP</option>
        <option value="jsp">JSP</option>
    </optgroup>
    </select>
    <script src="jquery.min.js"></script>
    <script src="fSelect.js"></script>
    <script>
        $(function() {
            console.log($('.sel').fSelect());
        });
    </script>
</body>
</html>

ZSH隐藏命令行前面的用户名和主机名

修改vim ~/.zshrc文件,在文件底部增加

隐藏用户名和主机名
prompt_context() {}
只保留用户名，隐藏主机名
prompt_context() {
  if [[ "$USER" != "$DEFAULT_USER" || -n "$SSH_CLIENT" ]]; then
    prompt_segment black default "%(!.%{%F{yellow}%}.)$USER"
  fi
}
只保留主机名，隐藏用户名
prompt_context() {
  if [[ "$USER" != "$DEFAULT_USER" || -n "$SSH_CLIENT" ]]; then
    prompt_segment black default "%(!.%{%F{yellow}%}.)$HOST"
  fi
}

修改后记得执行 source ~/.zshrc

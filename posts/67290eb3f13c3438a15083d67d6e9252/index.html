连接MySQL数据库

 package com.crlion.Test;
 ​
 import java.sql.*;
 ​
 public class Demo1 {
     public static void main(String[] args) {
         Connection connection = null;
         try {
 //        加载驱动
             Class.forName("com.mysql.cj.jdbc.Driver");
 //        创建连接
             connection = DriverManager.getConnection("jdbc:mysql:///test", "root", "root");
 //        预处理器
             PreparedStatement preparedStatement = connection.prepareStatement("select * from user");
 //        执行语句
             ResultSet resultSet = preparedStatement.executeQuery();
 //        处理结果
             while (resultSet.next()) {
                 int id = Integer.parseInt(resultSet.getString("id"));
                 String name = resultSet.getString("name");
                 String password = resultSet.getString("password");
                 System.out.println("id:"+id+" name:"+name+" password:"+password);
             }
         } catch (ClassNotFoundException | SQLException e) {
             e.printStackTrace();
         } finally {
             try {
 //        关闭连接
                 assert connection != null;
                 connection.close();
             } catch (SQLException e) {
                 e.printStackTrace();
             }
         }
     }
 }

预处理器

preparedStatement 和 statement的区别

联系：都是处理器对象，负责执行sql语句，statement是preparedStatement的父类，有更丰富的方法

区别：preparedStatement 会进行预处理可以防止sql注入，statement 不能防止sql注入

第三部分 事务的处理

事务的处理

 package com.crlion.Test;
 ​
 import java.sql.Connection;
 import java.sql.DriverManager;
 import java.sql.PreparedStatement;
 import java.sql.SQLException;
 ​
 public class Demo3 {
     public static void main(String[] args) {
         Connection connection = null;
         try {
 //        加载驱动
             Class.forName("com.mysql.cj.jdbc.Driver");
 //        创建连接
             connection = DriverManager.getConnection("jdbc:mysql:///test", "root", "root");
 //        设置不自动提交
             connection.setAutoCommit(false);
             PreparedStatement preparedStatement = connection.prepareStatement("update user set password = '666' where id='1'");
             preparedStatement.executeUpdate();
 //        写入一个异常
             System.out.println(1/0);
 ​
             preparedStatement = connection.prepareStatement("update user set password = '999' where id='2'");
             preparedStatement.executeUpdate();
             connection.commit();
         } catch (Exception e) {
             e.printStackTrace();
 //            发生异常执行回滚
             try {
                 assert connection != null;
                 connection.rollback();
             } catch (SQLException ex) {
                 ex.printStackTrace();
             }
         } finally {
             try {
 //        关闭连接
                 assert connection != null;
                 connection.close();
             } catch (SQLException e) {
                 e.printStackTrace();
             }
         }
     }
 }
 ​

结果1

执行异常语句，异常前，异常后的sql语句均未执行

 

 

结果2

不执行异常语句，两条sql语句执行成功

 System.out.println(1/0);

 

说明

两个sql语句组成一个事务

事务具有原子性，要么两个sql同时执行成功，要么同时执行失败

开始事务

 // 设置不自动提交 到提交事务之前可以写执行语句
 connection.setAutoCommit(false);

提交事务

 connection.commit();

回滚事务

事务执行失败，所有sql语句都不会生效

 connection.rollback();

 import org.springframework.transaction.TransactionStatus;
 import org.springframework.transaction.support.DefaultTransactionDefinition;
 ​
 import java.io.IOException;
 import java.util.Properties;
 ​
 public class JDBCTemplateUtils {
 ​
     private static String driverClassName="com.mysql.cj.jdbc.Driver";
     private static String jdbcUrl="jdbc:mysql:///test";
     private static String username="root";
     private static String password="root";
     private static HikariDataSource hikariDataSource;
 ​
     private static DataSourceTransactionManager dataSourceTransactionManager;
     static {
         Properties properties = new Properties();
         try {
             properties.load(JDBCTemplateUtils.class.getClassLoader().getResourceAsStream("jdbc.properties"));
         } catch (IOException e) {
             System.err.println("未找到配置文件，使用默认配置！");
             properties.setProperty("driverClassName",driverClassName);
             properties.setProperty("jdbcUrl",jdbcUrl);
             properties.setProperty("username",username);
             properties.setProperty("password",password);
         }
         HikariConfig hikariConfig = new HikariConfig(properties);
 //        创建连接池
         hikariDataSource = new HikariDataSource(hikariConfig);
 ​
     }
 ​
     /**
      * 获取连接对象
      * @return 连接对象
      */
     public static JdbcTemplate getJdbcTemplate(){
         return new JdbcTemplate(hikariDataSource);
     }
 ​
 ​
     /**
      * 初始化/获取 事务管理器
      */
     public static void getDataSourceTransactionManager(){
         if(dataSourceTransactionManager==null){
             dataSourceTransactionManager = new DataSourceTransactionManager(hikariDataSource);
         }else {
             System.err.println("已有事务管理器无需再创建！");
         }
     }
 ​
     /**
      * 开启/创建 事务
      * @return 事务
      */
     public static TransactionStatus beginTransaction(){
         DefaultTransactionDefinition defaultTransactionDefinition = new DefaultTransactionDefinition();
         return dataSourceTransactionManager.getTransaction(defaultTransactionDefinition);
     }
 ​
     /**
      * 提交事务
      * @param transaction 事务
      */
     public static void commitTransaction(TransactionStatus transaction){
         dataSourceTransactionManager.commit(transaction);
     }
 ​
     /**
      * 回滚事务
      * @param transaction 事务
      */
     public static void rollbackTransaction(TransactionStatus transaction){
         dataSourceTransactionManager.rollback(transaction);
     }
 }

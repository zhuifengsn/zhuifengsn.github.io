using namespace std ;
int n ;
int d,a[110] ;
int main(){
    scanf("%d%d",&n,&d) ;
    for (int i=1;i<=n;i++) scanf("%d",&a[i]) ;
    int ans=2;
    for (int i=2;i<=n;i++){
        if (a[i]-a[i-1]>=2*d+1) ans+=2 ;
        if (a[i]-a[i-1]==2*d) ans++ ;
    }
    printf("%d\n",ans) ;
}

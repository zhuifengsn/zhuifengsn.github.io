OpenCV Error: Assertion failed (size.width＞0 && size.height＞0) in imshow

QT Creator 配置完 OpenCV 之后用显示图片的demo测试时：

#include "mainwindow.h"
#include <QApplication>
#include <opencv2/opencv.hpp>
using namespace cv;

int main(int argc, char *argv[])
{
    Mat myImage = imread(":/left3.jpg"); //添加到了资源文件后+相对路径
    或者
    Mat myImage = imread("E:\\left3.jpg"); 放到E盘下的绝对路径+没有添加资源文件
    或者
    Mat myImage = imread("E:\\图片\\left3.jpg");放到E盘下的绝对路径+没有添加资源文件+路径含中文

    namedWindow("DEMO",CV_WINDOW_AUTOSIZE);
    imshow("DEMO",myImage);
    waitKey(0);
    return 0;
}

出现了如下报错：

OpenCV Error: Assertion failed (size.width>0 && size.height>0) in imshow, file E:\Opencv3.1.0\Windows\opencv\sources\modules\highgui\src\window.cpp, line 289
terminate called after throwing an instance of 'cv::Exception' what():  E:\Opencv3.1.0\Windows\opencv\sources\modules\highgui\src\window.cpp:289: error: (-215) size.width>0 && size.height>0 in function imshow

现归纳以下几点：

代码特征是否正确
Mat myImage = imread(“E:\MyQtProject\demo\left3.jpg”);添加到了资源文件后+绝对路径是
Mat myImage = imread(":/left3.jpg");添加到了资源文件后+相对路径否
Mat myImage = imread(“E:\left3.jpg”);放到E盘下的绝对路径+没有添加资源文件否
Mat myImage = imread(“E:\图片\left3.jpg”);放到E盘下的绝对路径+没有添加资源文件+路径含中文否

以下正确运行代码：

#include "mainwindow.h"
#include <QApplication>
#include <opencv2/opencv.hpp>
using namespace cv;

int main(int argc, char *argv[])
{
    Mat myImage = imread("E:\\MyQtProject\\demo\\left3.jpg"); 添加到了资源文件后+绝对路径
    namedWindow("DEMO",CV_WINDOW_AUTOSIZE);
    imshow("DEMO",myImage);
    waitKey(0);
    return 0;
}

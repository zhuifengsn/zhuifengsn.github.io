/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */


int kthToLast(struct ListNode* head, int k){
    struct ListNode* slow_p=head,*p=head;
    while(k--){
        p=p->next;
    }
    while(p!=NULL){
        slow_p=slow_p->next;
        p=p->next;
    }
    return slow_p->val;
}

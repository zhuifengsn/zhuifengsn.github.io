7俩个数字交换值

这道题虽然简单 但是很经典的一道题

#include <stdio.h>
int swap(int *p1,int *p2)
{
    int temp;
    temp=*p1;
    *p1=*p2;
    *p2=temp;
}
int main() 
{
   int a=1;
   int b=2;
   swap(&a,&b);
   printf("a:=%d\n",a);
   printf("b:=%d\n",b);
   return 0;
}

10数值比较 

这里题目的本意是通过if语句的 我们不使用if 我们来回顾一下回调函数的使用 我们利用qsort来完成数组的大小比较 不明白的参考笔记函数

#include<stdio.h>
#include<stdlib.h>
int comper (const void *p1,const void *p2)
{
    return *(int*)p1-*(int *)p2;
}
int main()
{
    int arr[5]={1,5,3,4,2};
    qsort(arr,sizeof(arr)/sizeof(arr[0]),sizeof(arr[0]),comper);
    for(int i=0;i<sizeof(arr)/sizeof(arr[0]);i++)
    {
        printf("%d\n",arr[i]);
    }
    return 0;      
}

14判断元音/辅音

判断输入的字母是元音，还是辅音。

英语有26个字母，元音只包括 a、e、i、o、u 这五个字母，其余的都为辅音。y是半元音、半辅音字母，但在英语中都把他当作辅音

#include<stdio.h>
int main()
{ 
    char num;
    scanf("%c",&num);
    if(num=='a'||num=='A'||num=='e'||num=='E'||num=='i'||num=='I'||num=='o'||num=='O'||num=='u'||num=='U')
    printf("%c是元音\n",num);
    else
    printf("%c不是元音\n",num);
	return 0;
}

16 判断闰年

#include <stdio.h>
 
int main()
{
    int year;
 
    printf("输入年份: ");
    scanf("%d",&year);
 
    if(year%4 == 0)
    {
        if( year%100 == 0)
        {
            // 这里如果被 400 整数是闰年
            if ( year%400 == 0)
                printf("%d 是闰年", year);
            else
                printf("%d 不是闰年", year);
        }
        else
            printf("%d 是闰年", year );
    }
    else
        printf("%d 不是闰年", year);
    
    return 0;
}

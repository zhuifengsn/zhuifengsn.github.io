#include<stdio.h>

int main(){
	int a,b;
	while(scanf("%d%d",&a,&b) != EOF){
	printf("%d\n",a+b);
	}
	return 0;
}

单组输入：

scanf("%d %d",&a,&b);

while(scanf("%d %d",&a,&b) != EOF){

}

while(n--){              //记得设置n值哦~

		printf("%d\n",a+ b);
	}

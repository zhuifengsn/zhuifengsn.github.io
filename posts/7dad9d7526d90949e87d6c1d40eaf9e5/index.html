def draw_point_cloud(result, batch=False):
    if not batch:
        for i in range(len(result)):
            o3d.visualization.draw_geometries(
                [result[i]],
                window_name="result",
                width=800, height=600,
                left=50, top=50,
                point_show_normal=False, mesh_show_back_face=False
            )
    else:
        o3d.visualization.draw_geometries(
            result,
            window_name="result",
            width=800, height=600,

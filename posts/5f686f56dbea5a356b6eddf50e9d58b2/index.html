vue跳转到新的页面之后，不让其回退的解决方法

直接把代码复制进去，方可实现！

  created(){
    // 跳转到登录界面之后，不让其回退。就直接添加下面这段代码即可实现
    history.pushState(null, null, document.URL);
    window.addEventListener("popstate", function () {
      history.pushState(null, null, document.URL);
    });
  }

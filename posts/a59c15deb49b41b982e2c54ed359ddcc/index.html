Status sort_by_expn(polynomial *P)
{
    polynomial s;
    polynomial p;
    polynomial q;
    int length;//ç»“ç‚¹ä¸ªæ•°
    length=PolynLength(*P);
    int i,j;
    for(i=0;i<length-1;i++){//å¤–å±‚å¾ªç¯
        j=length-1-i;
        s=*P;//æŒ‡å‘è¡¨å¤´
        p=(*P)->next;
        q=p->next;
        while(j--){//å†…å±‚å¾ªç¯
            if(q->elem.expn<p->elem.expn){
                //äº¤æ¢ç»“ç‚¹
                p->next=q->next;
                s->next=q;
                q->next=p;
            }
            s=s->next;
            p=s->next;
            q=p->next;
        }
    }
    return OK;
}

Status AddPolyn(polynomial *Pa,polynomial *Pb)
{
    /*
        a < b,è¿”å› -1
        a = b,è¿”å› 0
        a > b,è¿”å› 1
    */
    Node *pa;//Paçš„å·¥ä½œç»“ç‚¹
    Node *pb;//Pbçš„å·¥ä½œç»“ç‚¹
    Node *pa_prior;//å§‹ç»ˆæ˜¯paçš„ç›´æ¥å‰é©±
    Node *temp;//éœ€è¦é‡Šæ”¾çš„ç»“ç‚¹
    pa=(*Pa)->next;
    pb=(*Pb)->next;
    pa_prior=(*Pa);
    int flag;//è®°å½•cmpçš„ç»“æœ
    int e;//è®°å½•æŒ‡æ•°ç›¸åŒæ—¶ï¼Œç³»æ•°çš„å’Œ
    while(pa&&pb){//æœ‰ä¸€ä¸ªä¸ºç©ºå°±è·³å‡ºå¾ªç¯
        flag=cmp(pa->elem,pb->elem);
        if(flag==-1){//paå°
            pa=pa->next;
            pa_prior=pa_prior->next;
        }
        if(flag==0){//æŒ‡æ•°ç›¸åŒçš„é¡¹
            e=pa->elem.coef+pb->elem.coef;
            temp=pb;
            (*Pb)->next=pb->next;
            pb=(*Pb)->next;
            free(temp);
            if(e==0){
                //é‡Šæ”¾pa
                temp=pa;
                (*Pa)->next=pa->next;
                pa=(*Pa)->next;
                free(temp);
            }
            if(e!=0){
                pa->elem.coef=e;
            }
        }
        if(flag==1){//pbå°
            (*Pb)->next=pb->next;
            pb->next=pa;
            pa_prior->next=pb;
            pa_prior=pb;
            pb=(*Pb)->next;
        }
    }
    if(pb){
        pa_prior->next=pb;
        (*Pb)->next=NULL;
    }
    return OK;
}

Eurek注册中心配置

server:
  port: 8081
eureka:
  instance:
    hostname: localhost
  client:
    #是否向注册中心注册自己
    register-with-eureka: false
    #当前服务是否是注册中心，是的话不需要去检索服务；false表示是注册中心
    fetch-registry: false
      service-url:
        #eurek交互的地址查询服务和注册服务都需要这个地址
        defaultZone: http://${eureka.instance.hostname}:${server.port}/eureka/
